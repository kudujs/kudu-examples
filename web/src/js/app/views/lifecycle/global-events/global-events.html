Global Events

Here we show how to subscribe to global Kudu events. 
var kudu = require("kudu");
var $ = require("jquery");

$(kudu).on("viewFail", function(options) {
    alert("Sorry, an error occurred: ", options.error[0]);
});

$(kudu).on("viewRender", function(options) {
    console.log("view has been rendered");
});
