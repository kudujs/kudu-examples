Navigate Programatically

var kudu = require("kudu");
var myCtrl = require("myctrl");

kudu.go({
    ctrl: myCtrl,
    routeParams: {'id': 1, 'name': 'Bob'},
    args: {'myargs': ['one', 'two', 'three']}
});
